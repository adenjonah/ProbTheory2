\section*{\textcolor{red}{!! CRITICAL !!}}

\textbf{Gaussian} = \textbf{Normal} = N($\mu$,$\sigma^2$)

\textbf{Gaussian vector} = \textbf{MVN} = Multivariate Normal

\textbf{Independent components} = $\rho = 0$ = Independence (MVN!)

\textbf{Mean $\theta$} (Exp) $\Rightarrow$ $\lambda = 1/\theta$

$\psi(t)$ = MGF = $M_X(t)$

\section*{DISTRIBUTIONS}

\textbf{Normal}: Gaussian, N($\mu$,$\sigma^2$), Bell curve $\rightarrow$ Sec 3.3

\textbf{Standard Normal}: N(0,1), Z dist $\rightarrow$ Sec 3.3

\textbf{Bivariate Normal}: Gaussian vector, MVN, Jointly normal $\rightarrow$ Sec 4.5

\textbf{Exponential}: Exp($\lambda$), Memoryless, Waiting time $\rightarrow$ Sec 3.4

\textbf{Poisson}: Counting process, Arrival process, Rate $\lambda$ $\rightarrow$ Sec 2.3

\textbf{Binomial}: n trials, Success/failure $\rightarrow$ Sec 2.2

\textbf{Geometric}: First success, Trials until $\rightarrow$ Sec 2.4

\textbf{Lognormal}: $\ln X \sim N$, $e^X$ where $X \sim N$, Stock price $\rightarrow$ Sec 7.3

\textbf{Beta}: Beta($\alpha$,$\beta$), Conjugate prior $\rightarrow$ Sec 3.6

\textbf{Gamma}: Gamma($r$,$\lambda$), Sum of exp $\rightarrow$ Sec 3.5

\section*{PROCESSES}

\textbf{i.i.d.}: Independent identically distributed, Same distribution

\textbf{Arrival process}: Poisson process $\rightarrow$ Sec 2.3

\textbf{Waiting time}: Inter-arrival $\rightarrow$ Exponential

\textbf{Memoryless}: Exponential (cont.), Geometric (disc.)

\section*{OPERATIONS}

\textbf{Conditional on}: Given that, $|$, Restricting to

\textbf{Marginal}: Integrate out, Sum out $\rightarrow$ Sec 4.2

\textbf{Joint}: Together, Simultaneously $\rightarrow$ Sec 4.1

\textbf{Transformation}: Change of variable, Jacobian $\rightarrow$ Sec 4.6

\textbf{Sum of}: Convolution, MGF method $\rightarrow$ Sec 5.2

\section*{QUESTION KEYWORDS}

\textbf{``Large n''} $\rightarrow$ CLT (Sec 6.1)

\textbf{``Update belief''} $\rightarrow$ Bayesian (Sec 7.2)

\textbf{``Prior/Posterior''} $\rightarrow$ Bayesian (Sec 7.2)

\textbf{``Max/Min of n''} $\rightarrow$ Order Stats (Sec 4.7)

\textbf{``E[X|Y]''} $\rightarrow$ Cond. Expectation (Sec 7.1)

\section*{BAYESIAN}

\textbf{Prior}: $\pi(\theta)$, Initial belief, Before data

\textbf{Posterior}: $\pi(\theta|x)$, Updated belief, After data

\textbf{Likelihood}: $L(x|\theta)$, $P(\text{data}|\theta)$

\textbf{Conjugate}: Same family, Easy update $\rightarrow$ Beta-Binomial

\textbf{Predictive}: Future observation

\columnbreak

\section*{STATISTICS}

\textbf{Sample mean}: $\bar{X}$, $\bar{X}_n$, Average

\textbf{Order statistic}: $X_{(k)}$, k-th smallest

\textbf{Indicator}: $I_A$, $\mathbf{1}_A$, 1 if A else 0

\section*{LIMITS}

\textbf{CLT}: Central Limit Theorem, Normal approx

\textbf{LLN}: Law of Large Numbers

\textbf{Convergence}: $\xrightarrow{d}$ in distribution, $\xrightarrow{P}$ in probability

\section*{KEY FORMULAS}

\textbf{Covariance}: Cov(X,Y) = $E[XY] - E[X]E[Y]$

\textbf{Correlation}: $\rho$ = Cov$/\sigma_X\sigma_Y$

\textbf{Variance}: Var(X) = $E[X^2]-(E[X])^2$

\textbf{MGF}: $M_X(t) = E[e^{tX}]$ = $\psi(t)$

\section*{PROFESSOR NOTATION}

$\psi(t)$ = MGF

$g_1(x|y)$ = conditional PDF of X$|$Y

$\pi(\theta)$ = prior

$\pi(\theta|x)$ = posterior

$L(x|\theta)$ = likelihood

$H_i$ = hypothesis i

$\Phi(z)$ = std normal CDF

$z_\alpha$ = quantile

\section*{FINANCE}

\textbf{Stock price}: $S_t$, Lognormal, $S_0 e^Z$

\textbf{Log returns}: Normal, $\ln(S_t/S_0)$

\section*{\textcolor{blue}{NORMAL/GAUSSIAN}}

\subsection*{Single Variable}
$X \sim N(\mu, \sigma^2)$
\begin{enumerate}[leftmargin=*,itemsep=0pt,topsep=0pt]
\item Standardize: $Z = (X-\mu)/\sigma$
\item Use $\Phi$ table
\item $P(X > a) = 1 - \Phi((a-\mu)/\sigma)$
\end{enumerate}

\subsection*{Linear Transform}
$Y = aX + b$ where $X \sim N(\mu,\sigma^2)$\\
$\Rightarrow Y \sim N(a\mu+b, a^2\sigma^2)$

\subsection*{Sum of Normals}
$X \sim N(\mu_1,\sigma_1^2)$, $Y \sim N(\mu_2,\sigma_2^2)$ indep\\
$\Rightarrow X+Y \sim N(\mu_1+\mu_2, \sigma_1^2+\sigma_2^2)$

\subsection*{Bivariate Normal}
$(X,Y)$ jointly normal, correlation $\rho$:
\begin{itemize}[leftmargin=*,itemsep=0pt,topsep=0pt]
\item $aX+bY$ is normal
\item $\text{Var}(aX+bY) = a^2\sigma_X^2 + b^2\sigma_Y^2 + 2ab\rho\sigma_X\sigma_Y$
\item $\rho = 0 \Leftrightarrow$ independent (MVN only!)
\end{itemize}

\section*{\textcolor{blue}{CLT PROBLEMS}}

\subsection*{Standard Setup}
$X_1,...,X_n$ i.i.d., $E[X]=\mu$, $\text{Var}(X)=\sigma^2$

\begin{enumerate}[leftmargin=*,itemsep=0pt,topsep=0pt]
\item $\bar{X} \approx N(\mu, \sigma^2/n)$
\item $S_n = \sum X_i \approx N(n\mu, n\sigma^2)$
\item Standardize: $Z = \frac{\bar{X}-\mu}{\sigma/\sqrt{n}}$
\end{enumerate}

\subsection*{Game/Coin Problems}
\begin{enumerate}[leftmargin=*,itemsep=0pt,topsep=0pt]
\item Define $X_i$ = single game payoff
\item List PMF: $P(X_i = x)$
\item Calculate $\mu = E[X_i]$
\item Calculate $\sigma^2 = E[X^2] - \mu^2$
\item Total: $S_n \approx N(n\mu, n\sigma^2)$
\end{enumerate}

\subsection*{Finding n}
Want $P(\bar{X} > c) \geq p$:
$$n \geq \left(\frac{z^* \sigma}{c - \mu}\right)^2$$
where $z^*$ from $\Phi(z^*) = 1-p$

\columnbreak

\section*{\textcolor{blue}{BAYESIAN}}

\subsection*{Discrete Prior}
\begin{enumerate}[leftmargin=*,itemsep=0pt,topsep=0pt]
\item List $\theta_1, \theta_2, ...$
\item Priors: $P(\theta_i)$
\item Likelihoods: $P(\text{data}|\theta_i)$
\item Posterior: $\propto P(\text{data}|\theta_i)P(\theta_i)$
\item Normalize: sum = 1
\end{enumerate}

\subsection*{Monty Hall}
\textbf{Sober Monty:} Knows car location\\
- Switch wins 2/3\\

\textbf{Dizzy Monty:} Random choice\\
- No advantage

\subsection*{Conjugate Priors}
Beta($\alpha,\beta$) + Binomial(n,x) =\\
Beta($\alpha+x, \beta+n-x$)

\subsection*{Posterior Mean}
Beta($\alpha,\beta$): $E[\theta] = \frac{\alpha}{\alpha+\beta}$

\section*{\textcolor{blue}{LOGNORMAL}}

If $Y = e^X$ where $X \sim N(\mu,\sigma^2)$:
\begin{itemize}[leftmargin=*,itemsep=0pt,topsep=0pt]
\item $E[Y] = e^{\mu + \sigma^2/2}$
\item $P(Y > K) = P(X > \ln K)$
\item $P(Y \leq k) = \Phi\left(\frac{\ln k - \mu}{\sigma}\right)$
\end{itemize}

\subsection*{Stock Price}
$S = S_0 e^Z$ where $Z \sim N(r-\sigma^2/2, \sigma^2)$\\
$E[e^{-r}S] = S_0$ (risk-neutral!)

\subsection*{Product}
$X, Y$ lognormal indep\\
$XY$ is lognormal\\
$\ln(XY) = \ln X + \ln Y$

\columnbreak

\section*{\textcolor{blue}{ORDER STATISTICS}}

$X_1,...,X_n$ i.i.d. with CDF $F$

\subsection*{Maximum}
$$P(\max \leq a) = [F(a)]^n$$
$$P(\max > a) = 1 - [F(a)]^n$$

\subsection*{Minimum}
$$P(\min > a) = [1-F(a)]^n$$
$$P(\min \leq a) = 1 - [1-F(a)]^n$$

\subsection*{Uniform(0,1)}
$E[X_{(n)}] = n/(n+1)$\\
$E[X_{(1)}] = 1/(n+1)$

\section*{\textcolor{blue}{CONDITIONAL}}

\subsection*{Conditional Expectation}
$E[X|Y=y] = \int x f(x|y) dx$\\
$E[X] = E[E[X|Y]]$

\subsection*{Total Variance}
$\text{Var}(X) = E[\text{Var}(X|Y)] + \text{Var}(E[X|Y])$

\subsection*{BVN Conditional}
$Y|X=x \sim N(\mu_{Y|X}, \sigma^2_{Y|X})$\\
$\mu_{Y|X} = \mu_Y + \rho\frac{\sigma_Y}{\sigma_X}(x-\mu_X)$\\
$\sigma^2_{Y|X} = \sigma_Y^2(1-\rho^2)$

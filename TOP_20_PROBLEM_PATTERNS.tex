\begin{multicols}{3}

\subsection*{\textcolor{red}{1. CLT Game/Total}}
\textit{``n games'', ``total winnings''}\\
Find $E[X_i]$, Var$(X_i)$ $\rightarrow$ $S_n \approx N(n\mu, n\sigma^2)$\\
\textbf{Ref:} \S6.1, Template B

\subsection*{\textcolor{red}{2. Gaussian Vector}}
\textit{``Gaussian vector'', ``indep. components''}\\
$\rho=0 \Leftrightarrow$ independent (MVN), Cov = 0\\
\textbf{Ref:} \S4.5, Template A, G

\subsection*{\textcolor{red}{3. Lognormal Stock}}
\textit{``$S = S_0 e^Z$'', ``stock price''}\\
$E[e^X] = e^{\mu+\sigma^2/2}$\\
\textbf{Ref:} \S7.3, Template D

\subsection*{\textcolor{red}{4. Bayesian Discrete}}
\textit{``prior'', ``posterior'', ``defective''}\\
List $\theta_i$, priors, likelihoods $\rightarrow$ Bayes $\rightarrow$ Normalize\\
\textbf{Ref:} \S7.2, Template E

\subsection*{\textcolor{red}{5. BVN Conditional}}
\textit{``$Y|X=x$'', ``bivariate normal''}\\
$\mu_{Y|X} = \mu_Y + \rho\frac{\sigma_Y}{\sigma_X}(x-\mu_X)$\\
$\sigma^2_{Y|X} = \sigma_Y^2(1-\rho^2)$\\
\textbf{Ref:} \S4.5, Template F

\subsection*{\textcolor{red}{6. Exponential + CLT}}
\textit{``i.i.d. Exp'', ``mean $\theta$''}\\
\textbf{TRAP:} Mean $\theta \Rightarrow \lambda = 1/\theta$!\\
$\bar{X} \approx N(1/\lambda, 1/(n\lambda^2))$\\
\textbf{Ref:} \S3.4, \S6.1, Template C

\subsection*{\textcolor{red}{7. Monty Hall}}
\textit{``Monty Hall'', ``contestant picks''}\\
\textbf{Sober:} Switch wins 2/3\\
\textbf{Dizzy:} No advantage\\
\textbf{Ref:} \S8.1, Template J

\columnbreak

\subsection*{\textcolor{red}{8. Max/Min i.i.d.}}
\textit{``maximum of n'', ``minimum''}\\
$P(\max \leq a) = [F(a)]^n$\\
$P(\min > a) = [1-F(a)]^n$\\
\textbf{Ref:} \S4.7, Template L

\subsection*{9. Linear Combo BVN}
\textit{``$P(X+Y > c)$'', ``sum of jointly normal''}\\
$X+Y \sim N(\mu_X+\mu_Y, \sigma^2_{X+Y})$\\
$\sigma^2_{X+Y} = \sigma^2_X + \sigma^2_Y + 2\rho\sigma_X\sigma_Y$\\
\textbf{Ref:} \S4.5, \S8.3

\subsection*{10. Predictive Dist.}
\textit{``predict next'', ``posterior predictive''}\\
$P(X_{n+1}|\text{data}) = \sum_\theta P(X|\theta)P(\theta|\text{data})$\\
\textbf{Ref:} Template H

\subsection*{11. Product Lognormal}
\textit{``$XY$ where lognormal''}\\
$\ln(XY) = \ln X + \ln Y$ $\rightarrow$ Still lognormal!\\
\textbf{Ref:} Template I

\subsection*{12. BVN Parameters}
\textit{``find $\mu_Y, \sigma_Y, \rho$''}\\
Match $E[Y|X]$, Var$(Y|X)$ formulas\\
\textbf{Ref:} Template N

\subsection*{13. Ratio + CLT}
\textit{``$\bar{X}/(\bar{X}+c) < p$''}\\
Transform inequality first! Then CLT\\
\textbf{Ref:} Template O

\subsection*{14. Find n for CLT}
\textit{``smallest n such that''}\\
$n \geq (z^*\sigma/(c-\mu))^2$\\
\textbf{Ref:} Template K

\columnbreak

\subsection*{15. Confidence Int.}
\textit{``95\% CI'', ``confidence interval''}\\
$\bar{X} \pm z_{\alpha/2} \cdot \sigma/\sqrt{n}$, $z_{0.025} = 1.96$\\
\textbf{Ref:} \S6.4

\subsection*{16. Conjugate Prior}
\textit{``Beta-Binomial''}\\
Beta$(\alpha,\beta)$ + $x$ in $n$ $\Rightarrow$ Beta$(\alpha+x, \beta+n-x)$\\
\textbf{Ref:} \S7.2

\subsection*{17. Total Expectation}
\textit{``$E[X] = ?$'', condition on Y}\\
$E[X] = E[E[X|Y]]$\\
\textbf{Ref:} \S7.1, \S8.9

\subsection*{18. Total Variance}
\textit{``Var$(X)$'', condition on Y}\\
$\text{Var}(X) = E[\text{Var}(X|Y)] + \text{Var}(E[X|Y])$\\
\textbf{Ref:} \S7.1

\subsection*{19. Sum of Poisson}
\textit{``sum of Poisson''}\\
$X+Y \sim$ Poisson$(\lambda_1+\lambda_2)$\\
\textbf{Ref:} Template P

\subsection*{20. Transformation}
\textit{``find dist. of $Y=g(X)$''}\\
CDF: $F_Y(y) = P(g(X) \leq y)$\\
Jacobian: $f_Y = f_X|dx/dy|$\\
\textbf{Ref:} \S4.6

\end{multicols}

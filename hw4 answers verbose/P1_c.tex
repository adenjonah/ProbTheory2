\textbf{Problem 1(c): Find $E[X]$ and $\text{Var}(X)$.}

\textbf{Expected value $E[X]$:}
$$E[X] = \int_0^1 x \cdot f_X(x) \, dx = \int_0^1 x \cdot \frac{6x(2x + 1)}{7} \, dx$$
$$= \frac{6}{7} \int_0^1 x^2(2x + 1) \, dx = \frac{6}{7} \int_0^1 (2x^3 + x^2) \, dx$$
$$= \frac{6}{7} \left[ \frac{2x^4}{4} + \frac{x^3}{3} \right]_0^1 = \frac{6}{7} \left( \frac{1}{2} + \frac{1}{3} \right)$$
$$= \frac{6}{7} \cdot \frac{5}{6} = \frac{5}{7}$$

\textbf{Second moment $E[X^2]$:}
$$E[X^2] = \int_0^1 x^2 \cdot f_X(x) \, dx = \int_0^1 x^2 \cdot \frac{6x(2x + 1)}{7} \, dx$$
$$= \frac{6}{7} \int_0^1 x^3(2x + 1) \, dx = \frac{6}{7} \int_0^1 (2x^4 + x^3) \, dx$$
$$= \frac{6}{7} \left[ \frac{2x^5}{5} + \frac{x^4}{4} \right]_0^1 = \frac{6}{7} \left( \frac{2}{5} + \frac{1}{4} \right)$$
$$= \frac{6}{7} \cdot \frac{13}{20} = \frac{39}{70}$$

\textbf{Variance:}
$$\text{Var}(X) = E[X^2] - (E[X])^2 = \frac{39}{70} - \left(\frac{5}{7}\right)^2$$
$$= \frac{39}{70} - \frac{25}{49} = \frac{39 \cdot 7 - 25 \cdot 10}{490} = \frac{273 - 250}{490} = \frac{23}{490}$$

\textbf{Verification:} $E[X] = 0.714$ ✓, $\text{Var}(X) = 0.0469$ ✓

% Problem 5: Bayesian posterior with uniform prior

\textbf{Given:} Proportion $\theta$ of defective items unknown. Prior: $\theta \sim \text{Uniform}[0, 1]$. Data: 8 items selected, 3 defective. Find the posterior distribution, expected mean, and variance.

\textbf{Solution:}

\textbf{Prior:} $\xi(\theta) = 1$ for $\theta \in [0, 1]$ (equivalently, $\theta \sim \text{Beta}(1, 1)$)

\textbf{Likelihood:} $P(\text{3 defective out of 8} \mid \theta) = \binom{8}{3} \theta^3 (1-\theta)^5 \propto \theta^3 (1-\theta)^5$

\textbf{Posterior:}
$$f(\theta \mid \text{data}) \propto \xi(\theta) \cdot L(\theta) \propto 1 \cdot \theta^3 (1-\theta)^5 = \theta^3 (1-\theta)^5$$

Recognizing this as the Beta kernel with $\alpha = 4$ and $\beta = 6$:
$$\theta^3 (1-\theta)^5 = \theta^{(4-1)} (1-\theta)^{(6-1)}$$

Therefore:
$$\boxed{\theta \mid \text{data} \sim \text{Beta}(4, 6)}$$

\textbf{Posterior pdf:}
$$f(\theta \mid \text{data}) = \frac{1}{B(4,6)} \theta^3 (1-\theta)^5 = 504 \, \theta^3 (1-\theta)^5 \quad \text{for } \theta \in [0, 1]$$

\textbf{Posterior Mean:}
$$E[\theta \mid \text{data}] = \frac{\alpha}{\alpha + \beta} = \frac{4}{4 + 6} = \boxed{\frac{2}{5} = 0.4}$$

\textbf{Posterior Variance:}
$$\text{Var}(\theta \mid \text{data}) = \frac{\alpha \beta}{(\alpha + \beta)^2 (\alpha + \beta + 1)} = \frac{4 \times 6}{10^2 \times 11} = \frac{24}{1100} = \boxed{\frac{6}{275} \approx 0.0218}$$

\textbf{Verification:} $B(4, 6) = \frac{3! \cdot 5!}{9!} = \frac{720}{362880} = \frac{1}{504}$, so $\frac{1}{B(4,6)} = 504$. $\checkmark$

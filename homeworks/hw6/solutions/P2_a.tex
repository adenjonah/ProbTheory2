% Problem 2(a): Dice prediction - first roll

\textbf{Setup:} Five dice with 4, 6, 8, 12, or 20 sides. One chosen uniformly at random.

\textbf{Prior predictive probability that first roll is 7:}

Only dice with $\geq 7$ sides can roll a 7. Using the law of total probability:
\begin{align*}
P(x_1 = 7) &= \sum_{d} P(x_1 = 7 \mid d\text{-sided}) \cdot P(d\text{-sided}) \\
&= \frac{1}{5}\left(0 + 0 + \frac{1}{8} + \frac{1}{12} + \frac{1}{20}\right) \\
&= \frac{1}{5} \cdot \frac{15 + 10 + 6}{120} = \frac{1}{5} \cdot \frac{31}{120} = \boxed{\frac{31}{600}}
\end{align*}

\textbf{Posterior probabilities for the chosen die after first roll = 7:}

Using Bayes' theorem with prior $P(d) = 1/5$ for each die:
\begin{center}
\begin{tabular}{|c|c|c|c|c|}
\hline
Die & Prior & Likelihood $P(7|d)$ & Prior $\times$ Lik & Posterior \\
\hline
4-sided & $\frac{1}{5}$ & $0$ & $0$ & $0$ \\
6-sided & $\frac{1}{5}$ & $0$ & $0$ & $0$ \\
8-sided & $\frac{1}{5}$ & $\frac{1}{8}$ & $\frac{1}{40}$ & $\frac{15}{31}$ \\
12-sided & $\frac{1}{5}$ & $\frac{1}{12}$ & $\frac{1}{60}$ & $\frac{10}{31}$ \\
20-sided & $\frac{1}{5}$ & $\frac{1}{20}$ & $\frac{1}{100}$ & $\frac{6}{31}$ \\
\hline
\end{tabular}
\end{center}

\textbf{Posterior predictive probability that second roll is 7:}
\begin{align*}
P(x_2 = 7 \mid x_1 = 7) &= \sum_d P(x_2 = 7 \mid d) \cdot P(d \mid x_1 = 7) \\
&= \frac{1}{8} \cdot \frac{15}{31} + \frac{1}{12} \cdot \frac{10}{31} + \frac{1}{20} \cdot \frac{6}{31} \\
&= \frac{1}{31}\left(\frac{15}{8} + \frac{10}{12} + \frac{6}{20}\right) = \frac{1}{31} \cdot \frac{225 + 100 + 36}{120} = \boxed{\frac{361}{3720}}
\end{align*}

\textbf{Verification:} $\frac{15}{31} + \frac{10}{31} + \frac{6}{31} = 1$. $\checkmark$

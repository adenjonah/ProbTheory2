\textbf{Problem 1(b): Find the marginal cumulative distribution functions $F_X$ and $F_Y$ and the marginal pdf $f_X$ and $f_Y$.}

\textbf{Marginal pdf of $X$:}
$$f_X(x) = \int_0^1 f(x,y) \, dy = \int_0^1 \frac{12}{7}(x^2 + xy) \, dy$$
$$= \frac{12}{7} \int_0^1 (x^2 + xy) \, dy = \frac{12}{7} \left[ x^2y + \frac{xy^2}{2} \right]_0^1$$
$$= \frac{12}{7} \left( x^2 + \frac{x}{2} \right) = \frac{12x}{7} \left( x + \frac{1}{2} \right) = \frac{6x(2x + 1)}{7}$$

\textbf{Marginal pdf of $Y$:}
$$f_Y(y) = \int_0^1 f(x,y) \, dx = \int_0^1 \frac{12}{7}(x^2 + xy) \, dx$$
$$= \frac{12}{7} \int_0^1 (x^2 + xy) \, dx = \frac{12}{7} \left[ \frac{x^3}{3} + \frac{x^2y}{2} \right]_0^1$$
$$= \frac{12}{7} \left( \frac{1}{3} + \frac{y}{2} \right) = \frac{4}{7} + \frac{6y}{7}$$

\textbf{Marginal cdf of $X$:}
$$F_X(x) = \int_0^x f_X(u) \, du = \int_0^x \frac{6u(2u + 1)}{7} \, du$$
$$= \frac{6}{7} \int_0^x (2u^2 + u) \, du = \frac{6}{7} \left[ \frac{2u^3}{3} + \frac{u^2}{2} \right]_0^x$$
$$= \frac{6}{7} \left( \frac{2x^3}{3} + \frac{x^2}{2} \right) = \frac{6x^2(4x + 3)}{21} = \frac{2x^2(4x + 3)}{7}$$

\textbf{Marginal cdf of $Y$:}
$$F_Y(y) = \int_0^y f_Y(v) \, dv = \int_0^y \left( \frac{4}{7} + \frac{6v}{7} \right) dv$$
$$= \frac{1}{7} \int_0^y (4 + 6v) \, dv = \frac{1}{7} \left[ 4v + 3v^2 \right]_0^y$$
$$= \frac{4y + 3y^2}{7} = \frac{y(4 + 3y)}{7}$$

\textbf{Verification:} $\int_0^1 f_X(x) \, dx = 1.0$ ✓, $\int_0^1 f_Y(y) \, dy = 1.0$ ✓

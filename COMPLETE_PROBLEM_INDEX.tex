\subsection{\hwproblem{Homework 2}}

\subsubsection{HW2-Q1: Polya Urn Model}

\textbf{Problem:} Box with $r$ red, $b$ blue balls. Draw, return with $k$ additional of same color. Find $P(\text{first 3 red, 4th blue})$.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{1.2} - Conditional Probability
\item \formularef{Chain rule: $P(A_1 \cap \ldots \cap A_n) = P(A_1)P(A_2|A_1)\cdots$}
\end{itemize}

\textbf{Solution:}
\[P = \frac{r}{r+b} \cdot \frac{r+k}{r+b+k} \cdot \frac{r+2k}{r+b+2k} \cdot \frac{b}{r+b+3k}\]

\hrulefill

\subsubsection{HW2-Q7: Five Coins (Bayesian)}

\textbf{Problem:} Five coins with $p_i = 0, 1/4, 1/2, 3/4, 1$. Select coin randomly, observe head. Find posterior probabilities.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{7.2} - Bayesian Statistics
\item \sectionref{1.3} - Bayes' Theorem
\item \formularef{$P(\text{coin}_i | H) = \frac{P(H|\text{coin}_i)P(\text{coin}_i)}{\sum_j P(H|\text{coin}_j)P(\text{coin}_j)}$}
\end{itemize}

\textbf{Key Steps:}
\begin{enumerate}
\item Prior: $P(\text{coin}_i) = 1/5$
\item Likelihood: $P(H|\text{coin}_i) = p_i$
\item Posterior: Apply Bayes formula
\end{enumerate}

\hrulefill

\textbf{Variations:}

\item Different test accuracy $\rightarrow$ Sensitivity analysis
\item Multiple tests $\rightarrow$ Sequential updating
\end{itemize}

\subsection{\hwproblem{Homework 3}}

\subsubsection{HW3-Q1: Joint PMF}

\textbf{Problem:} $p_{X,Y}(x,y) = c(x^2 + y^2)$ for $x \in \{1,2,4\}$, $y \in \{1,3\}$.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{4.1} - Joint Distributions
\item \sectionref{4.2} - Marginals
\item \sectionref{4.4} - Covariance
\end{itemize}

\textbf{Solution Steps:}
\begin{enumerate}
\item Find $c$: $\sum\sum p(x,y) = 1$
\item Marginals: $p_X(x) = \sum_y p(x,y)$
\item Expectations: $E[X] = \sum x \cdot p_X(x)$
\item Variance: $\text{Var}(X) = E[X^2] - E[X]^2$
\end{enumerate}

\hrulefill

\subsubsection{HW3-Q5: Reward for HT Patterns}

\textbf{Problem:} Coin tosses with $P(H) = p$. Reward 1 for each HT pattern. Find $E[R]$, $\text{Var}(R)$.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{7.4} - Indicator Random Variables
\item \formularef{$R = \sum I_i$ where $I_i = 1$ if HT at position $i$}
\end{itemize}

\textbf{Key Formulas:}
\begin{align*}
E[R] &= (n-1) \cdot p(1-p) \\
\text{Var}(R) &= \sum \text{Var}(I_i) + 2\sum \text{Cov}(I_i, I_j)
\end{align*}

\hrulefill

\subsubsection{HW3-Q6: Exponential Distribution}

\textbf{Problem:} $X \sim \text{Exp}(\lambda)$. Find survival function, moments, distribution of $\min(X_1, X_2)$.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{3.4} - Exponential Distribution
\item \formularef{$P(X > x) = e^{-\lambda x}$}
\end{itemize}

\warning{If ``mean $\theta = 3$'' then $\lambda = 1/3$ NOT $\lambda = 3$!}

\textbf{Key Results:}
\begin{itemize}
\item $E[X] = 1/\lambda$, $\text{Var}(X) = 1/\lambda^2$
\item $\min(X_1, X_2) \sim \text{Exp}(\lambda_1 + \lambda_2)$
\item Memoryless: $P(X > s+t | X > s) = P(X > t)$
\end{itemize}

\subsection{\hwproblem{Homework 4}}

\subsubsection{HW4-Q1: Joint PDF Analysis}

\textbf{Problem:} $f(x,y) = c(x^2 + xy)$ on $[0,1]^2$. Find $c$, marginals, covariance, check independence.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{4.1} - Joint Distributions
\item \sectionref{4.2} - Marginal/Conditional
\item \sectionref{4.4} - Covariance/Correlation
\item \sectionref{4.3} - Independence Check
\end{itemize}

\textbf{Solution:}
\begin{enumerate}
\item $c = 12/7$ (integrate to 1)
\item $f_X(x) = \frac{12}{7}(x^2 + x/2)$
\item $E[X] = 5/7$, $\text{Var}(X) \approx 0.0469$
\item $\text{Cov}(X,Y) \approx -0.0034$
\item NOT independent (joint $\neq$ product of marginals)
\end{enumerate}

\hrulefill

\subsubsection{HW4-Q4: Meeting Problem}

\textbf{Problem:} $A, B \sim \text{Uniform}(0,60)$ independent. Find $P(|A-B| < 15)$.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{4.1} - Joint Distributions
\item \sectionref{3.2} - Uniform Distribution
\item \template{U17} (Meeting Problem)
\end{itemize}

\textbf{Key Method:} Draw square $[0,60]^2$, shade region where condition holds, compute area fraction.

\textbf{Solution:} $P(|A-B| < 15) = 1 - (45/60)^2 = 1 - 9/16 = 7/16$

\hrulefill

\subsubsection{HW4-Q5: Overlapping Sums}

\textbf{Problem:} $X_i \sim \text{Exp}(2)$ i.i.d. $X = \sum_{i=1}^n X_i$, $Y = \sum_{i=n-7}^{2n-8} X_i$. Find $\text{Cov}(X,Y)$.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{4.4} - Covariance
\item \formularef{$\text{Cov}(\sum X_i, \sum X_j) = \sum_{\text{overlap}} \text{Var}(X_i)$}
\end{itemize}

\textbf{Solution:} Overlap has 8 terms. $\text{Cov}(X,Y) = 8 \cdot \text{Var}(X_i) = 8 \cdot (1/4) = 2$.

\subsection{\hwproblem{Homework 5}}

\subsubsection{HW5-Q1: Bivariate Normal}

\textbf{Problem:} $(X,Y)$ jointly normal with $\mu_X = 1$, $\sigma_X^2 = 2$, $\mu_Y = -2$, $\sigma_Y^2 = 3$, $\rho = -2/3$.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{4.5} - Bivariate Normal
\item \template{A} (Gaussian Vector Problems)
\item \template{F} (BVN Conditional)
\item \template{G} (Linear Combination Independence)
\end{itemize}

\textbf{Part (a): $P(X + Y > 0)$}
\begin{enumerate}
\item $X + Y$ is normal
\item $E[X+Y] = 1 + (-2) = -1$
\item $\text{Var}(X+Y) = 2 + 3 + 2(-2/3)\sqrt{6} \approx 1.73$
\item Standardize and use $\Phi$ table
\end{enumerate}

\textbf{Part (b): Find $a$ for independence}
\begin{enumerate}
\item Set $\text{Cov}(aX+Y, X+2Y) = 0$
\item For MVN: $\text{Cov} = 0 \Leftrightarrow$ independent
\end{enumerate}

\hrulefill

\subsubsection{HW5-Q2: Lognormal Distribution}

\textbf{Problem:} $X$ lognormal with parameters $\mu = 3$, $\sigma^2 = 1.44$. Find $P(X \leq 6.05)$.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{7.3} - Lognormal Distribution
\item \formularef{$P(X \leq x) = \Phi\left(\frac{\ln x - \mu}{\sigma}\right)$}
\end{itemize}

\textbf{Solution:}
\[P(X \leq 6.05) = \Phi\left(\frac{\ln(6.05) - 3}{1.2}\right) = \Phi\left(\frac{1.8 - 3}{1.2}\right) = \Phi(-1) \approx 0.159\]

\hrulefill

\subsubsection{HW5-Q4: Conditional BVN}

\textbf{Problem:} Test scores bivariate normal. If score on A is 80, find $P(B > 90)$.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{4.5} - Bivariate Normal
\item \template{F} (BVN Conditional)
\item \formularef{$Y|X=x \sim N\left(\mu_Y + \rho\frac{\sigma_Y}{\sigma_X}(x-\mu_X), (1-\rho^2)\sigma_Y^2\right)$}
\end{itemize}

\begin{itemize}

\item \template{K} (Finding n for CLT)

\begin{align*}
P\left(|Z| < \frac{0.3\sqrt{n}}{3}\right) &\geq 0.95 \\
\frac{0.3\sqrt{n}}{3} &\geq 1.96 \\
\sqrt{n} &\geq 19.6 \\
n &\geq 385
\end{align*}

\subsubsection{PF-Q1: Coin Game with CLT}

\textbf{Problem:} Win \$3 if HH, lose \$1 if TT, \$0 otherwise. Play 400 times.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{6.1} - CLT
\item \template{B} (CLT Game Problems)
\end{itemize}

\textbf{Solution:}
\begin{enumerate}
\item Single game: $E[X] = 3(1/4) - 1(1/4) = 0.5$, $\text{Var}(X) = 2.25$
\item Total: $S_{400} \sim N(200, 900)$
\item $P(S > 240) = P(Z > 4/3) = 1 - \Phi(1.33) \approx 0.092$
\end{enumerate}

\hrulefill

\subsubsection{PF-Q2: Gaussian Vector Independence}

\textbf{Problem:} $X_1, X_2$ i.i.d. $N(0,1)$. $Y_1 = aX_1 + X_2$, $Y_2 = X_1 + bX_2$. Find $a, b$ for independence.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{4.5} - Bivariate Normal
\item \template{A} (Gaussian Vector)
\item \template{G} (Linear Combination Independence)
\end{itemize}

\textbf{Key Insight:} For MVN, independence $\Leftrightarrow$ $\text{Cov} = 0$.

\textbf{Solution:} $\text{Cov}(Y_1, Y_2) = a + b = 0 \Rightarrow b = -a$.

\hrulefill

\subsubsection{PF-Q3: Exponential CLT}

\textbf{Problem:} $X_1, \ldots, X_{100}$ i.i.d. $\text{Exp}$ with mean 3. Find $P\left(\frac{\bar{X}}{\bar{X}+3} < 0.5\right)$.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{6.1} - CLT
\item \sectionref{3.4} - Exponential
\item \template{C} (Exponential + CLT)
\item \template{O} (Ratio with Sample Mean)
\end{itemize}

\warning{Mean $\theta = 3$ means $\lambda = 1/3$!}

\textbf{Solution:}
\begin{enumerate}
\item Transform: $\frac{\bar{X}}{\bar{X}+3} < 0.5 \Leftrightarrow \bar{X} < 3$
\item By CLT: $\bar{X} \approx N(3, 9/100)$
\item $P(\bar{X} < 3) = \Phi(0) = 0.5$
\end{enumerate}

\hrulefill

\subsubsection{PF-Q4: Lognormal Stock Price}

\textbf{Problem:} $S = S_0 e^Z$ where $Z \sim N(r-\sigma^2/2, \sigma^2)$. Find $E[e^{-r}S]$ and $P(S > 100)$.

\textbf{Primary Solution Path:}
\begin{itemize}
\item \sectionref{7.3} - Lognormal
\item \template{D} (Lognormal Stock Price)
\end{itemize}

\textbf{Key Formula:} $E[e^X] = e^{\mu + \sigma^2/2}$ for $X \sim N(\mu, \sigma^2)$.

\textbf{Solution (a):} $E[e^{-r}S] = S_0 = 100$ (risk-neutral pricing result)

\textbf{Solution (b):} $P(S > 100) = P(Z > 0) = 1 - \Phi\left(\frac{-(r-\sigma^2/2)}{\sigma}\right)$

\hrulefill

\section{Formula Quick Reference}

\textbf{Probability:}
\begin{align*}
P(A|B) &= \frac{P(A \cap B)}{P(B)} & P(A) &= \sum P(A|B_i)P(B_i) \\
P(H|E) &= \frac{P(E|H)P(H)}{P(E)} & &\text{(Bayes)}
\end{align*}

\textbf{Expectation/Variance:}
\begin{align*}
E[aX+b] &= aE[X] + b & \text{Var}(aX+b) &= a^2\text{Var}(X) \\
\text{Var}(X) &= E[X^2] - E[X]^2 & \text{Cov}(X,Y) &= E[XY] - E[X]E[Y]
\end{align*}

\textbf{CLT:}
\[Z = \frac{\bar{X} - \mu}{\sigma/\sqrt{n}} \xrightarrow{d} N(0,1)\]

\textbf{Bivariate Normal Conditional:}
\[Y|X=x \sim N\left(\mu_Y + \rho\frac{\sigma_Y}{\sigma_X}(x-\mu_X), (1-\rho^2)\sigma_Y^2\right)\]

\textbf{Lognormal:}
\[E[e^X] = e^{\mu + \sigma^2/2} \quad \text{for } X \sim N(\mu, \sigma^2)\]

\textbf{Beta-Binomial Conjugate:}
\[\text{Beta}(\alpha, \beta) + k \text{ successes in } n \rightarrow \text{Beta}(\alpha+k, \beta+n-k)\]
